      <label className="block text-sm font-medium text-muted-strong">
        Imagens do quarto
        <input
          type="file"
          accept="image/*"
          multiple
          onChange={(event: ChangeEvent<HTMLInputElement>) => {
            const files = event.target.files ? Array.from(event.target.files) : [];
            setCreateImages(files);
          }}
          className="mt-2 block w-full cursor-pointer rounded-lg border border-dashed border-slate-300 bg-slate-50 px-3 py-6 text-center text-sm text-muted transition hover:border-primary hover:text-primary dark:border-slate-700 dark:bg-slate-900/60 dark:text-slate-200"
        />
        {createImages.length > 0 && (
          <ul className="mt-2 space-y-1 text-xs text-muted">
            {createImages.map((file) => (
              <li key={file.name}>{file.name}</li>
            ))}
          </ul>
        )}
      </label>
